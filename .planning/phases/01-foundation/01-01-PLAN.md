  <action>
    Add to schema.prisma:
    ```prisma
    model User {
      id           String   @id @default(cuid())
      username     String   @unique
      passwordHash String   @map("password_hash")
      isActive     Boolean  @default(true) @map("is_active")
      roleId       String   @map("role_id")
      role         Role     @relation(fields: [roleId], references: [id])
      createdAt    DateTime @default(now()) @map("created_at")
      updatedAt    DateTime @updatedAt @map("updated_at")
      createdBy    String?  @map("created_by") // foreign key to user who created this user

      auditLogs      AuditLog[]
      refreshTokens  RefreshToken[]

      @@index([roleId])
      @@map("users")
    }
    ```
  </action>
  <verify>grep -q "model User" prisma/schema.prisma && grep -q "passwordHash" prisma/schema.prisma</verify>

    ```prisma
    model AuditLog {
      id         String   @id @default(cuid())
      userId     String   @map("user_id")
      user       User     @relation(fields: [userId], references: [id])
      action     String   // LOGIN, LOGOUT, USER_CREATE, USER_EDIT, USER_DEACTIVATE, ROLE_ASSIGN

      entityId   String?  @map("entity_id") // ID of affected resource, null for system actions
      createdAt  DateTime @default(now()) @map("created_at")

      @@index([userId])
      @@index([createdAt])
      @@map("audit_log")
    }
    ```

    ```typescript
        update: {},
        create: { name: 'SUPERADMIN' }
      })

      const finance = await prisma.role.upsert({
        where: { name: 'FINANCE' },
        update: {},
        create: { name: 'FINANCE' }

        where: { name: 'CASHIER' },
        update: {},
        create: { name: 'CASHIER' }
      })

    }

    main()
      .catch(console.error)
      .finally(() => prisma.$disconnect())
    ```
